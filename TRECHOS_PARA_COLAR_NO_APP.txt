# 1) No topo do arquivo, junto dos imports:
import os
import json

# 2) Antes do botão "Iniciar geração", inserir:

st.markdown("### Complementar localização manual de fornecedores")

fornecedores_manuais = st.text_area(
    "Se algum fornecedor não for localizado automaticamente, informe um por linha exatamente como aparece no PDF.",
    placeholder="Exemplo:\nACÁCIA COMÉRCIO DE MEDICAMENTOS LTDA\nMEDILAR IMPORTACAO E DISTRIBUICAO DE PRODUTOS MEDICO HOSPITALARES S/A\nZION MEDPHARMA IMPORTACAO, EXPORTACAO COMERCIO E SERVICOS LTDA",
)

gerar_mesmo_com_divergencia = st.checkbox(
    "Se ainda houver divergência, gerar mesmo assim apenas com os fornecedores encontrados.",
    value=False,
)

# 3) Dentro do if generate_clicked, ANTES de chamar run_engine(...), inserir:

nomes_manuais = [
    linha.strip()
    for linha in fornecedores_manuais.splitlines()
    if linha.strip()
]
os.environ["LICITASUITE_FORNECEDORES_MANUAIS"] = json.dumps(nomes_manuais, ensure_ascii=False)
os.environ["LICITASUITE_GERAR_MESMO_COM_DIVERGENCIA"] = "SIM" if gerar_mesmo_com_divergencia else "NAO"
